function Editor() {
  const [model, setModel] = useState(defaultModel);
  const [svg, setSvg] = useState(null);

  async function uploadImage(file) {
    const form = new FormData();
    form.append("image", file);

    const res = await fetch("/api/trace", {
      method: "POST",
      body: form
    });

    const data = await res.json();
    setSvg(data.svg);
  }

  return (
    <div className="grid grid-cols-2">
      <Upload onChange={uploadImage} />
      <ThreePreview svg={svg} model={model} />
      <Controls model={model} setModel={setModel} />
      <PublishButton svg={svg} model={model} />
    </div>
  );
}

🧷 4. PREVIEW + RENDER PIPELINE
function ThreePreview({ svg, model }) {
  const ref = useRef();

  useEffect(() => {
    if (!svg) return;

    const scene = new THREE.Scene();
    const mesh = generateGeometry(svg, model);

    scene.add(mesh);

    ref.current.scene = scene;
  }, [svg, model]);

  return <canvas ref={ref} />;
}
🖼️ 5. PREVIEW IMAGE GENERATION (Marketplace requirement)

Used for listings.

function renderPreview(mesh) {
  const renderer = new THREE.WebGLRenderer({ preserveDrawingBuffer: true });
  renderer.setSize(512, 512);

  const scene = new THREE.Scene();
  scene.add(mesh);

  const camera = new THREE.PerspectiveCamera(75, 1, 0.1, 1000);
  camera.position.z = 80;

  renderer.render(scene, camera);

  return renderer.domElement.toDataURL("image/png");
}

app.post("/api/order", async (req, res) => {
  const { listingId } = req.body;

  const listing = await db.listing.find(listingId);
  const design = await db.design.find(listing.designId);

  const mesh = generateGeometry(design.svgSource, design.paramModel);

  const stl = new THREE.STLExporter().parse(mesh);

  const fileUrl = await uploadToStorage(stl);

  const order = await db.order.create({
    listingId,
    stlFileUrl: fileUrl,
    status: "queued"
  });

  enqueuePrint(order.id);

  res.json(order);
});
